## Install

```bash
npm install --save login-component
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'login-component'
import 'login-component/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
